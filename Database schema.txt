create database onlineexam

create table Admin (
	admin_id int primary key not null identity(1,1),
	username nvarchar(50) not null,
	password nvarchar(50) not null,
);

create table Users (
	user_id int primary key not null identity(1,1),
	phone nvarchar(10) not null,
	name nvarchar(50) not null,
	email nvarchar(50) not null,
	password nvarchar(50) not null,
	college nvarchar(50) not null,
	year_of_passing int not null,
	qualification nvarchar(50) not null,
	city nvarchar(50) not null
);

create table Tests (
	test_id int primary key not null identity(1,1),
	subject_name nvarchar(10) not null,
	test_date date not null,
	duration int not null,
	L_one_req int not null,
	L_two_req int not null,
	L_three_req int not null,
	admin_id int not null foreign key references Admin(admin_id)
);

create table Attempts (
	attempt_id int primary key not null identity(1,1),
	user_id int not null foreign key references Users(user_id),
	test_id int not null foreign key references Tests(test_id),
	level_cleared int not null,
	L_one_marks int not null default 0,
	L_two_marks int not null default 0,
	L_three_marks int not null default 0,
);

create table Questions (
	question_id int primary key not null identity(1,1),
	test_id int not null foreign key references Tests(test_id),
	question nvarchar(1000) not null ,
	level_id int not null,
	options_one nvarchar(100) not null,  
	options_two nvarchar(100) not null,  
	options_three nvarchar(100) not null,  
	options_four nvarchar(100) not null,  
	options_correct nvarchar(100) not null,  
);
create table Questionsadvanced (
	question_id int primary key not null identity(1,1),
	test_id int not null foreign key references Tests(test_id),
	is_single_correct int not null ,--1 if it is single valued ans,0 if otherwise
	question nvarchar(1000) not null ,
	level_id int not null,
	options_one nvarchar(100) not null,  
	options_two nvarchar(100) not null,  
	options_three nvarchar(100) not null,  
	options_four nvarchar(100) not null,  
	options_correct nvarchar(100) not null,  
);
